## 函数指针本质——三分钟重温

```c
// 函数名 = 函数的入口地址（和数组名类似）
void hello(void) { printf("hello\n"); }
printf("%p\n", hello);   // 打印函数地址
printf("%p\n", &hello);  // 等价，&可省略

// 定义一个函数指针
void (*fp)(void);        // fp是指针，指向 void fn(void) 类型的函数
fp = hello;              // 赋值
fp();                    // 调用（等价于 hello()）
```

记忆法则：把函数声明里的函数名换成 `(*指针名)`，就得到了函数指针的声明。

```
原函数:   void hello(void)
函数指针: void (*fp)(void)
          把 hello 换成 (*fp)
```

## typedef 化简——面试写代码的必杀技

不用 typedef 的原生写法又长又丑：

```c
// 原生：声明一个函数指针变量
void (*handler)(int, char*);

// 原生：函数指针数组（已经读不动了）
void (*handlers[5])(int, char*);

// 原生：返回函数指针的函数（面试官都可能写错）
void (*get_handler(void))(int, char*);
```

用 typedef 彻底拯救可读性：

```c
// 1. 先 typedef 出类型
typedef void (*handler_t)(int, char*);

// 2. 然后和普通类型一样用
handler_t h1;                // 函数指针变量
handler_t handlers[5];       // 函数指针数组
handler_t get_handler(void); // 返回函数指针的函数
```

### typedef 公式（面试手写不出错）

```
typedef  返回值类型 (*新类型名)(参数列表);

例：
typedef int  (*cmp_t)(const void*, const void*);   // qsort 比较函数
typedef void (*task_t)(void*);                     // FreeRTOS 任务函数
typedef void (*isr_t)(void);                       // 中断服务函数
```

## 实战一：状态机——嵌入式最核心的设计模式

### 不用函数指针的状态机（if-else 地狱）

```c
typedef enum { STATE_IDLE, STATE_RUN, STATE_ERROR, STATE_STOP } state_t;

void state_machine(state_t s, int event) {
    switch (s) {
        case STATE_IDLE:
            if (event == EVT_START) { ... }
            else if (event == EVT_TIMEOUT) { ... }
            break;
        case STATE_RUN:
            if (event == EVT_STOP) { ... }
            else if (event == EVT_ERROR) { ... }
            break;
        // 每加一个状态，switch 就胖一圈
    }
}
```

### 用函数指针重构（状态表驱动）

```c
// ==== 底层驱动代码（不随状态增多而膨胀）====

typedef enum { STATE_IDLE, STATE_RUN, STATE_ERROR, STATE_COUNT } state_t;
typedef enum { EVT_START, EVT_STOP, EVT_ERROR, EVT_TIMEOUT, EVT_COUNT } event_t;

// 每个状态是一个函数
typedef void (*state_handler_t)(void);

// 状态转移表：state_table[当前状态][事件] = 下一状态
state_t state_table[STATE_COUNT][EVT_COUNT] = {
    // EVT_START    EVT_STOP    EVT_ERROR   EVT_TIMEOUT
    {  STATE_RUN,   STATE_IDLE, STATE_ERROR, STATE_IDLE },  // IDLE
    {  STATE_RUN,   STATE_IDLE, STATE_ERROR, STATE_RUN  },  // RUN
    {  STATE_IDLE,  STATE_IDLE, STATE_IDLE, STATE_IDLE },   // ERROR
};

// 状态动作表
state_handler_t action_table[STATE_COUNT] = {
    idle_handler, run_handler, error_handler
};

// 主循环（永不加 if-else）
state_t current_state = STATE_IDLE;

void state_machine(event_t event) {
    action_table[current_state]();                         // 执行当前状态的动作
    current_state = state_table[current_state][event];     // 查表跳转
}
```

### 面试追问：状态表放 Flash（嵌入式特有优化）

```c
// STM32 上，大表默认放 SRAM 太奢侈，加 const 强制放 Flash
static const state_t state_table[STATE_COUNT][EVT_COUNT] = {
    { STATE_RUN, STATE_IDLE, STATE_ERROR, STATE_IDLE },
    // ...
};

// 同理，函数指针本身在 Flash（代码段），但函数指针表需要 const
static void (* const action_table[STATE_COUNT])(void) = {
    idle_handler, run_handler, error_handler
};
//       ^^^^^ const 修饰指针表本身（指针指向的代码本来就在 Flash，不需要 const）
```

## 实战二：回调函数——解耦的终极武器

### 场景：硬件驱动层通知应用层

```c
// ==== 驱动层（drv_uart.h）====
typedef void (*uart_rx_callback_t)(uint8_t byte);   // 回调类型定义

void drv_uart_register_rx_callback(uart_rx_callback_t cb);
// 驱动层不知道也不关心 cb 指向什么函数，只管收到数据时调用

// ==== 驱动层（drv_uart.c）====
static uart_rx_callback_t rx_cb = NULL;

void drv_uart_register_rx_callback(uart_rx_callback_t cb) {
    rx_cb = cb;
}

void USART1_IRQHandler(void) {   // 中断服务函数
    if (USART_GetITStatus(USART1, USART_IT_RXNE)) {
        uint8_t byte = USART_ReceiveData(USART1);
        if (rx_cb != NULL) {
            rx_cb(byte);         // 回调：通知上层"收到一个字节"
        }
    }
}

// ==== 应用层（app.c）====
void on_uart_byte(uint8_t byte) {
    // 应用层处理：协议解析、命令匹配...
    if (byte == 0xAA) { /* 帧头 */ }
}

void app_init(void) {
    drv_uart_register_rx_callback(on_uart_byte);  // 注册回调
}
```

### 面试进阶：带用户参数的回调（void* 妙用）

```c
// 问题：一个驱动被多个模块共用，如何区分谁的回调？
// 答案：回调签名里带一个 void* 参数

typedef void (*timer_cb_t)(void *user_data);

typedef struct {
    timer_cb_t callback;
    void      *user_data;
} timer_entry_t;

void timer_register(timer_cb_t cb, void *user_data);

// 应用层：
void led_timer_cb(void *data) {
    int led_id = *(int*)data;    // 知道是哪个 LED
    GPIO_Toggle(led_pins[led_id]);
}

int led1 = 1, led2 = 2;
timer_register(led_timer_cb, &led1);   // LED1 的回调
timer_register(led_timer_cb, &led2);   // LED2 的回调，同一个函数
```

这正是 FreeRTOS 软件定时器回调 `void (*TimerCallbackFunction_t)(TimerHandle_t xTimer)` 的简化版——`xTimer` 就是这个 `void*` 的具象化。

## 实战三：命令表——你的闹钟项目里无处不在

### 基础版：菜单/命令分发

```c
typedef void (*cmd_handler_t)(void *arg);

typedef struct {
    const char   *name;       // 命令字符串
    cmd_handler_t handler;     // 处理函数
    const char   *help;        // 帮助信息（嵌入式常省略，Flash省空间）
} cmd_entry_t;

// ==== 命令表（编译期确定，放 Flash）====
static const cmd_entry_t cmd_table[] = {
    { "start",  start_cmd,  "启动系统" },
    { "stop",   stop_cmd,   "停止系统" },
    { "status", status_cmd, "查询状态" },
    { "reset",  reset_cmd,  "复位设备" },
    { NULL,     NULL,       NULL }       // 哨兵值，标记表尾
};

// ==== 命令解析（不会随命令增多而膨胀）====
void cmd_parse(const char *input) {
    for (int i = 0; cmd_table[i].name != NULL; i++) {
        if (strcmp(input, cmd_table[i].name) == 0) {
            cmd_table[i].handler(NULL);
            return;
        }
    }
    printf("Unknown command\n");
}
```

### 进阶：用函数指针做"弱绑定"默认实现

```c
// 场景：库函数提供一个默认的 assert 处理，但用户可覆盖

void __attribute__((weak)) assert_failed_handler(const char *file, int line) {
    // 默认实现：死循环
    while (1);
}

// 用户代码：
void assert_failed_handler(const char *file, int line) {
    printf("ASSERT: %s:%d\n", file, line);
    // 记录日志后复位
    NVIC_SystemReset();
}
// 链接时用户的强符号覆盖库的弱符号
```

## 实战四：qsort——C标准库里的函数指针教科书

```c
// qsort 原型
void qsort(void *base, size_t nmemb, size_t size,
           int (*compar)(const void *, const void *));
//         ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
//         这就是一个函数指针参数

// ==== 四个例子 ====

// 1. 排序 int 数组
int cmp_int(const void *a, const void *b) {
    return *(int*)a - *(int*)b;
}
qsort(arr, 5, sizeof(int), cmp_int);

// 2. 排序字符串（按字典序）
int cmp_str(const void *a, const void *b) {
    return strcmp(*(char**)a, *(char**)b);
}
qsort(strs, 3, sizeof(char*), cmp_str);

// 3. 排序结构体（按年龄）
typedef struct { char name[32]; int age; } person_t;
int cmp_age(const void *a, const void *b) {
    return ((person_t*)a)->age - ((person_t*)b)->age;
}
qsort(people, 10, sizeof(person_t), cmp_age);

// 4. 降序排列——改一下比较符号就行
int cmp_int_desc(const void *a, const void *b) {
    return *(int*)b - *(int*)a;
}
```

面试常问：qsort 的实现一般用什么排序？（答：多数实现用快排，小数据量回退到插入排序）

## 函数指针与指针函数的区别（面试必考混淆点）

```
函数指针：指针，指向函数
  int (*fp)(int, int);     // fp 是函数指针

指针函数：函数，返回指针
  int *fn(int, int);        // fn 是函数，返回值是 int*
```

```c
// 函数指针变量
int (*operation)(int, int);
operation = add;    // operation 指向 add 函数

// 指针函数定义
int *find_max(int *arr, int n) {  // 返回 int* 的函数
    int *max = &arr[0];
    for (int i = 1; i < n; i++)
        if (arr[i] > *max) max = &arr[i];
    return max;
}
```

记忆口诀：括号包住谁，谁就是本质。
- `(*fp)` → 括号包住指针 → 是指针 → 函数指针
- `fn()` → 括号是函数调用 → 是函数 → 指针函数

## 常见笔试题陷阱

### 陷阱1：函数指针声明和赋值类型不匹配

```c
void handler(int x) { ... }
void (*fp)(void);            // 参数列表：void
fp = handler;                // ❌ 类型不兼容！void(int) vs void(void)
// 编译器至少给 warning
```

### 陷阱2：通过函数指针调用时忘了判空

```c
typedef void (*cb_t)(int);
cb_t callback = NULL;       // 某处忘记注册
callback(42);               // ❌ 空指针解引用，直接 HardFault

// 正确姿势
if (callback) callback(42);
// 或预埋一个空函数当默认值
```

### 陷阱3：返回局部变量地址（和指针篇联动）

```c
int *get_array(void) {
    int local[10] = {0};
    return local;           // ❌ 返回栈上地址，函数返回后失效
}

void (*get_handler(void))(int) {
    void handler(int x) { ... }
    return handler;         // ✓ 函数代码在 .text 段，生命周期永久
}
// 函数指针指向的代码不会消失，但若是嵌套函数（GCC扩展），注意栈帧
```

## 嵌入式面试模拟：函数指针三板斧

**第一问**："用函数指针实现一个简单的按键消抖状态机。"

```c
typedef enum { BTN_IDLE, BTN_WAIT, BTN_PRESSED } btn_state_t;
typedef btn_state_t (*btn_handler_t)(uint8_t level);

btn_state_t idle_handler(uint8_t level) {
    return (level == 0) ? BTN_WAIT : BTN_IDLE;
}
btn_state_t wait_handler(uint8_t level) {
    static int count = 0;
    if (level == 0) {
        count++;
        return (count > 5) ? BTN_PRESSED : BTN_WAIT;
    }
    count = 0;
    return BTN_IDLE;
}
// ... 主循环里查表调用
```

**第二问**："你的温湿度闹钟里哪些地方用了函数指针？"

> 指针篇里的状态机示例就是这题的答案——多模式切换、按键处理分发。面试时要能随手在白板上画出状态表和 action_table。

**第三问**："FreeRTOS 任务创建为什么传一个函数指针而不是直接调函数？"

> 因为 FreeRTOS 需要接管任务的栈、优先级、调度——它不是简单调用，而是在任务自己的上下文里执行。传指针让调度器在合适的时机"跳过去"执行，而不是当场执行。
